sdk_example.py 
# Use Anthropic SDK — any model via JBridge
import anthropic

client = anthropic.Anthropic(
    api_key="jb-xxx",
    base_url="https://api.jbridge.ai"
)

message = client.messages.create(
    model="gpt-5.4",
    max_tokens=1024,
    messages=[
        {
            "role": "user",
            "content": "Explain quantum entanglement"
        }
    ]
)

print(message.content[0].text)
# Switch to deepseek-v3.2 by changing model=""
# — everything else stays identical
